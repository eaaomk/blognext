            // æ ¹æ®æ—¶é—´æŸ¥æ‰¾æ–°æ¶ˆæ¯åœ¨é“¾è¡¨ä¸­çš„ä½ç½®
            for (;;) {
                prev = p;
                p = p.next;
                if (p == null || when < p.when) {
                    break;
                }
                //å½“éå†åˆ°é“¾è¡¨å°¾éƒ¨ï¼Œæˆ–è€…æ‰¾åˆ°åˆé€‚çš„ä½ç½®
                if (needWake && p.isAsynchronous()) {
                    needWake = false;
                }
            }
            //æ’å…¥æ–°æ¶ˆæ¯åˆ°é“¾è¡¨ä¸­
            msg.next = p; // invariant: p == prev.next
            prev.next = msg;
        }
        //æ ¹æ®æƒ…å†µå”¤é†’äº‹ä»¶é˜Ÿåˆ—
        // We can assume mPtr != 0 because mQuitting is false.
        if (needWake) {
            nativeWake(mPtr);
        }
    }
    return true;
}
